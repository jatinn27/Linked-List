// Middle of linked List
// https://www.geeksforgeeks.org/problems/finding-middle-element-in-a-linked-list/1

class Solution{
    public:
    /* Should return data of middle node. If linked list is empty, then  -1*/
    int getMiddle(Node *head)
    {
        if(head == NULL){
            return -1;
        }
        
        if(head ->next == NULL){
            return head->data;
        }
        
        Node* slow = head;
        Node* fast = head;
        
        while(fast!=NULL){
            fast = fast ->next ;
            if(fast!=NULL){
                fast = fast-> next;
                slow = slow ->next;
            }
        }
        
        return slow->data;
    }
};
